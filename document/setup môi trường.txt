hệ điều hành android tối thiểu 4.0.3
update sdk (quan trọng nhất là phần extras có gói nào up hết gói đó)
sau khi chuẩn bị môi trường thì sẽ k chạy được do không dùng được google api nên cần import thư viện vào eclipse
vào đường dẫn ...\sdk\extras\google\google_play_services\libproject
import vào (nhớ chọn copy to workspace)
add thư viện vào project đang làm
chỉnh sửa trong file mainifest
<?xml version="1.0" encoding="utf-8"?>
<manifest xmlns:android="http://schemas.android.com/apk/res/android"
    package="healthcarepro.steprun"
    android:versionCode="1"
    android:versionName="1.0" >

    <uses-sdk
        android:minSdkVersion="14"
        android:targetSdkVersion="21" />
    <uses-permission
        android:name="android.permission.WRITE_EXTERNAL_STORAGE" />
    <uses-permission android:name="android.permission.GET_ACCOUNTS" />
    <uses-permission android:name="android.permission.INTERNET" />
    <uses-permission android:name="android.permission.USE_CREDENTIALS" />

    <application
        android:allowBackup="true"
        android:icon="@drawable/ic_launcher"
        android:label="@string/app_name"
        android:theme="@style/AppTheme" >
        <activity
            android:name="healthcarepro.steprun.HealthCareMain"
            android:label="@string/app_name" >
            <intent-filter>
                <action android:name="android.intent.action.MAIN" />

                <category android:name="android.intent.category.LAUNCHER" />
            </intent-filter>
        </activity>
        <service
            android:name="healthcarepro.steprun.GoogleFitService"
            android:enabled="true" >
        </service>
        <meta-data android:name="com.google.android.gms.version" android:value="@integer/google_play_services_version" />
    </application>

</manifest>
